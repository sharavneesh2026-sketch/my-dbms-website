P31Function to return Nth highest salary
CREATE OR REPLACE FUNCTION f12(n NUMBER) RETURN NUMBER IS
    v_sal NUMBER;
BEGIN
    SELECT sal INTO v_sal FROM (
        SELECT sal, DENSE_RANK() OVER (ORDER BY sal DESC) rnk
        FROM scott.emp
    ) WHERE rnk = n AND ROWNUM = 1;
    RETURN v_sal;
END;
/
-- Execute:
SELECT f12(2) FROM DUAL;  -- 2nd highest
📟 DBMS_OUTPUT / Result
3000 (2nd highest salary)
