P30Salary increment based on job using FOR UPDATE cursor
DECLARE
    CURSOR c1 IS SELECT job, sal FROM scott.emp FOR UPDATE;
BEGIN
    FOR rec IN c1 LOOP
        IF rec.job = 'MANAGER' THEN
            UPDATE scott.emp SET sal = sal * 1.20
            WHERE CURRENT OF c1;
        ELSIF rec.job = 'CLERK' THEN
            UPDATE scott.emp SET sal = sal * 1.10
            WHERE CURRENT OF c1;
        END IF;
    END LOOP;
END;
